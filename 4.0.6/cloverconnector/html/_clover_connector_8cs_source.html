159  public void Sale(SaleRequest request)
160  {
161  lastRequest = request ?? new SaleRequest();
162 
163  if (deviceObserver == null || merchantInfo == null)
164  {
165  throw new Exception("InitializeConnection() has not been called on Clover Connector");
166  }
167  if (Device == null || !IsReady)
168  {
169  deviceObserver.onFinishCancel(ResponseCode.ERROR,
170  TxType.SALE,
171  "Device Connection Error",
172  "In Sale : SaleRequest - The Clover device is not connected.");
173  return;
174  }
175  if (request == null)
176  {
177  deviceObserver.onFinishCancel(ResponseCode.FAIL,
178  TxType.SALE,
179  "Request Validation Error",
180  "In Sale : SaleRequest - The request that was passed in for processing is empty.");
181  return;
182  }
183  if (String.IsNullOrEmpty(request.ExternalId))
184  {
185  deviceObserver.onFinishCancel(ResponseCode.FAIL,
186  TxType.SALE,
187  "Request Validation Error",
188  "In Sale : SaleRequest - The request ExternalId cannot be null or blank. Original Request = " + request);
189  return;
190  }
191  if (request.Amount <= 0)
192  {
193  deviceObserver.onFinishCancel(ResponseCode.FAIL,
194  TxType.SALE,
195  "Request Validation Error",
196  "In Sale : SaleRequest - The request amount cannot be zero. Original Request = " + request);
197  return;
198  }
199  if (request.TipAmount != null && request.TipAmount < 0)
200  {
201  deviceObserver.onFinishCancel(ResponseCode.FAIL,
202  TxType.SALE,
203  "Request Validation Error",
204  "In Sale : SaleRequest - The request tip amount cannot be less than zero. Original Request = " + request);
205  return;
206  }
207  if (request.VaultedCard != null && !merchantInfo.supportsVaultCards)
208  {
209  deviceObserver.onFinishCancel(ResponseCode.UNSUPPORTED,
210  TxType.SALE,
211  "Merchant Configuration Validation Error",
212  "In Sale : SaleRequest - Vault Card support is not offered by the merchant configured gateway. Original Request = " + request);
213  return;
214  }
215 
216  PayIntent payIntent = new PayIntent();
217  payIntent.externalPaymentId = request.ExternalId;
218  payIntent.externalReferenceId = request.ExternalReferenceId;
219  payIntent.transactionType = PayIntent.TransactionType.PAYMENT;
220  payIntent.amount = request.Amount;
221  payIntent.tipAmount = request.TipAmount ?? 0;
222  payIntent.taxAmount = request.TaxAmount;
223  payIntent.vaultedCard = request.VaultedCard;
224  payIntent.requiresRemoteConfirmation = true;
225 
226  if (request.Extras != null && request.Extras.Count > 0)
227  {
228  payIntent.passThroughValues = request.Extras;
229  }
230  if (request.CardNotPresent.HasValue && request.CardNotPresent.Value)
231  {
232  payIntent.isCardNotPresent = true;
233  }
234  if (request.PresentQrcOnly.HasValue)
235  {
236  payIntent.isPresentQrcOnly = request.PresentQrcOnly;
237  }
238 
239  TransactionSettings ts = getTransactionRequestOverrides(request);
240  ts.tippableAmount = request.TippableAmount;
241  if (request.DisableCashback.HasValue && request.DisableCashback.Value)
242  {
243  ts.disableCashBack = true;
244  }
245  if (request.ForceOfflinePayment.HasValue)
246  {
247  ts.forceOfflinePayment = request.ForceOfflinePayment;
248  }
249  if (request.AllowOfflinePayment.HasValue)
250  {
251  ts.allowOfflinePayment = request.AllowOfflinePayment;
252  }
253  if (request.ApproveOfflinePaymentWithoutPrompt.HasValue)
254  {
255  ts.approveOfflinePaymentWithoutPrompt = request.ApproveOfflinePaymentWithoutPrompt;
256  }
257  if (request.TipMode.HasValue)
258  {
259  ts.tipMode = request.TipMode.GetAltTipMode();
260  }
261  if (request.TippableAmount.HasValue)
262  {
263  ts.tippableAmount = request.TippableAmount.Value;
264  }
265  payIntent.transactionSettings = ts;
266  Device.doTxStart(payIntent, null, TxType.SALE);
267  }
